rmath.linalg

Matrix decompositions and solvers powered by faer. All operations accept and return rmath.Array.

example.py
import rmath as rm

A = rm.Array([[4, 7], [2, 6]])
b = rm.Array([[1], [2]])
x = rm.linalg.solve(A, b)

Q, R = rm.linalg.qr(A)
U, S, Vt = rm.linalg.svd(A)

solvers.py
import rmath as rm

A = rm.Array([[4, 7], [2, 6]])
b = rm.Array([[1], [2]])

# Solve Ax = b
x = rm.linalg.solve(A, b)
print(x)  # solution vector

# Determinant and inverse
print(rm.linalg.det(A))   # 10.0
A_inv = rm.linalg.inv(A)
print(A @ A_inv)          # ≈ identity matrix
decompositions.py
import rmath as rm

A = rm.Array([[1, 2], [3, 4]])

# QR decomposition
Q, R = rm.linalg.qr(A)
print(Q @ R)  # ≈ A (reconstruction)

# SVD: A = U · diag(S) · Vt
U, S, Vt = rm.linalg.svd(A)
print(S)  # singular values as Vector

# Symmetric eigenvalue decomposition
sym = rm.Array([[2, 1], [1, 3]])
vals, vecs = rm.linalg.eigh(sym)
print(vals)   # eigenvalues
print(vecs)   # eigenvectors (columns)
