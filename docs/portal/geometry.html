rmath.geometry

Distance metrics, 3D rotations via quaternions, and computational geometry algorithms. All vector operations are parallelized.

example.py
import rmath as rm

v1 = rm.Vector([1, 2, 3])
v2 = rm.Vector([4, 5, 6])

# Distance and similarity
dist = rm.geometry.euclidean_distance(v1, v2)
sim = rm.geometry.cosine_similarity(v1, v2)

# 3D Vector cross product
cross = rm.geometry.cross_product(v1, v2)

Quaternion

Unit quaternion for 3D rotations: q = w + xi + yj + zk.

q1 * q2QuaternionHamilton product (rotation composition)
quaternion.py
import rmath as rm

# Create a 90-degree rotation around the Z-axis
import math
half_angle = math.pi / 4  # 45 degrees in rad
q = rm.geometry.Quaternion(math.cos(half_angle), 0, 0, math.sin(half_angle))

v = rm.Vector([1.0, 0.0, 0.0])
rotated_v = q.rotate_vector(v)
print(rotated_v.to_list())  # ≈ [0.0, 1.0, 0.0]

distances.py
import rmath as rm

a = rm.Vector([0.0, 3.0])
b = rm.Vector([4.0, 0.0])

print(rm.geometry.euclidean_distance(a, b))  # 5.0  (3-4-5 triangle)
print(rm.geometry.manhattan_distance(a, b))  # 7.0  (|0-4| + |3-0|)
print(rm.geometry.cosine_similarity(a, b))   # 0.0  (orthogonal)

computational.py
import rmath as rm

# A simple square polygon
poly_x = [0.0, 2.0, 2.0, 0.0]
poly_y = [0.0, 0.0, 2.0, 2.0]

print(rm.geometry.is_point_in_polygon(1.0, 1.0, poly_x, poly_y)) # True
print(rm.geometry.is_point_in_polygon(3.0, 3.0, poly_x, poly_y)) # False
