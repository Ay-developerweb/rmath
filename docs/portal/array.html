Core Module

rmath.array

N-dimensional array engine. Flat contiguous Vec<f64> storage with a separate shape vector. All element-wise operations parallelized via Rayon when N ≥ 8,192. Supports 1-D, 2-D, and arbitrary N-D shapes.

example.py
import rmath as rm

a = rm.Array([[1, 2], [3, 4]])
b = rm.Array.eye(2)

c = a @ b          # matrix multiply
d = a.sin().exp()  # element-wise chain
avg = a.mean()     # global reduction

constructors.py
import rmath as rm

# From nested lists
a = rm.Array([[1, 2, 3], [4, 5, 6]])
print(a)
# Array([
#        [ 1.0000,  2.0000,  3.0000],
#        [ 4.0000,  5.0000,  6.0000]])

# N-dimensional constructors
mat  = rm.Array.zeros(3, 4)           # 2-D: shape [3, 4]
vol  = rm.Array.ones(2, 3, 4)        # 3-D: shape [2, 3, 4]
batch = rm.Array.randn(16, 3, 256, 256) # 4-D: shape [16, 3, 256, 256]

# Identity and filled
I = rm.Array.eye(3)                 # 3×3 identity
f = rm.Array.full(2, 3, 7.0)       # 2×3 filled with 7.0

# Ranges
r = rm.Array.arange(0, 10, 2)        # [0, 2, 4, 6, 8]
l = rm.Array.linspace(0, 1, 5)      # [0.0, 0.25, 0.5, 0.75, 1.0]

shape_access.py
import rmath as rm

a = rm.Array([[1, 2, 3], [4, 5, 6]])

print(a.shape)       # [2, 3]
print(a.size())       # 6
print(a.nrows())      # 2
print(a.ncols())      # 3
print(a.get(0, 2))    # 3.0
print(a.row(1))       # [4.0, 5.0, 6.0]
print(a.col(0))       # [1.0, 4.0]

# Reshape (no data copy)
flat = a.reshape([6])       # shape [6]
mat  = a.reshape([3, 2])    # shape [3, 2]

# Transpose
t = a.transpose()
print(t.shape)  # [3, 2]

Vector ↔ Array Conversion

Move data freely between Vector (1-D) and Array (N-D) without leaving Rust.

MethodDirectionDescription
Array(vector)Vector → ArrayWrap a Vector as a 1-D Array
arr.into_vector()Array → VectorFlatten a 1-D or 2-D Array into a Vector
arr.to_flat_list()Array → listFlat Python list from any dimension
arr.reshape([n])N-D → 1-DReshape to 1-D without copy
vector_array.py
import rmath as rm

# ── Vector → Array ──
v = rm.Vector([1, 2, 3, 4, 5, 6])
a = rm.Array(v)              # 1-D Array, shape [6]
mat = a.reshape([2, 3])    # reshape into 2×3 matrix
print(mat)
# Array([
#        [ 1.0000,  2.0000,  3.0000],
#        [ 4.0000,  5.0000,  6.0000]])

# ── Array → Vector ──
back = mat.into_vector()
print(back)  # Vector([1, 2, 3, 4, 5, 6], len=6)

# ── Use Vector tools on Array data ──
data = rm.Array.randn(100, 10)
flat = data.into_vector()

# Now you get Vector's unique features
sorted_data = flat.sort()
top_10 = flat.sort_desc().head(10)
outliers = flat.filter_gt(2.0)  # elements > 2 std devs

# ── Move back to Array for matrix ops ──
clean = rm.Array(sorted_data).reshape([100, 10])

interop_example.py
import numpy as np
import rmath as rm

# NumPy → RMath → NumPy roundtrip
np_arr = np.array([[1.0, 2.0], [3.0, 4.0]])
rm_arr = rm.Array.from_numpy(np_arr)
back   = rm_arr.to_numpy()
assert np.allclose(np_arr, back)  # True

# Works directly with sklearn (no manual conversion)
from sklearn.linear_model import LinearRegression
model = LinearRegression()
model.fit(rm.Array.randn(50, 1), rm.Array.randn(50, 1))
